# Variable global para recordar quien inicio sesion durante todo el programa
usuario_actual_id = None

# tabla para guardar los datos personales del alumno
class Usuario(Base_Datos):
    __tablename__ = "usuarios_registrados"
    
    id = Column(String(36), primary_key=True, default=lambda: str(uuid.uuid4()))
    nombre = Column(String, nullable=False)
    correo = Column(String, unique=True, nullable=False)
    codigo_alumno = Column(String, unique=True, nullable=False)
    contrasena = Column(String, nullable=False)
    semestre = Column(String, nullable=False)

# tabla para las clases, vinculada al alumno
class ClaseDelHorario(Base_Datos):
    __tablename__ = "mis_clases_guardadas"
    
    id = Column(String(36), primary_key=True, default=lambda: str(uuid.uuid4()))
    usuario_id = Column(String(36), ForeignKey("usuarios_registrados.id"))
    materia = Column(String, nullable=False)
    dia = Column(String) 
    profesor = Column(String)
    contacto_profesor = Column(String, default="[]") # Guardamos formato JSON: [{"tipo":"Correo", "valor":"..."}]
    hora_inicio = Column(DateTime(timezone=True))
    hora_fin = Column(DateTime(timezone=True))
    salon = Column(String)

async def preparar_base_de_datos():
    async with motor_base_datos.begin() as conexion:
        await conexion.run_sync(Base_Datos.metadata.create_all)

async def registrar_nuevo_alumno():
    print("\n--- REGISTRO DE NUEVO ALUMNO ---")
    nombre_input = input("Ingresa tu nombre completo: ")
    correo_input = input("Ingresa tu correo electronico: ")
    codigo_input = input("Ingresa tu codigo de alumno: ")
    contra_input = input("Crea una contrasena: ")
    semestre_input = input("Ingresa tu semestre actual: ")
    
    async with crear_sesion() as sesion_bd:
        nuevo_usuario = Usuario(
            nombre=nombre_input,
            correo=correo_input,
            codigo_alumno=codigo_input,
            contrasena=contra_input,
            semestre=semestre_input
        )
        sesion_bd.add(nuevo_usuario)
        await sesion_bd.commit()
        await sesion_bd.refresh(nuevo_usuario)
        
        print("\nRegistro exitoso. Bienvenido, " + nuevo_usuario.nombre)
        return nuevo_usuario.id

async def iniciar_sesion():
    print("\n--- INICIAR SESION ---")
    identificador = input("Ingresa tu correo o tu codigo de alumno: ")
    contra_input = input("Ingresa tu contrasena: ")
    
    async with crear_sesion() as sesion_bd:
        consulta = select(Usuario).where(
            ((Usuario.correo == identificador) | (Usuario.codigo_alumno == identificador)) & 
            (Usuario.contrasena == contra_input)
        )
        resultado = await sesion_bd.execute(consulta)
        usuario_encontrado = resultado.scalar_one_or_none()
        
        if usuario_encontrado:
            print("\nSesion iniciada correctamente. Hola de nuevo, " + usuario_encontrado.nombre)
            return usuario_encontrado.id
        else:
            print("\nError: Datos incorrectos. Intentalo de nuevo.")
            return None

async def guardar_clase_en_postgres(nombre_materia: str, dia: str, inicio_fecha_hora: str, fin_fecha_hora: str, nombre_salon: str, nombre_maestro: str) -> str:
    global usuario_actual_id
    try:
        fecha_inicio_real = datetime.fromisoformat(inicio_fecha_hora.replace('Z', '+00:00'))
        fecha_fin_real = datetime.fromisoformat(fin_fecha_hora.replace('Z', '+00:00'))
        
        async with crear_sesion() as sesion_bd:
            nueva_clase = ClaseDelHorario(
                usuario_id=usuario_actual_id,
                materia=nombre_materia,
                dia=dia,
                profesor=nombre_maestro if nombre_maestro else "Sin maestro",
                contacto_profesor="[]", # Inicializa como lista JSON vacía
                hora_inicio=fecha_inicio_real,
                hora_fin=fecha_fin_real,
                salon=nombre_salon if nombre_salon else "Sin salon"
            )
            sesion_bd.add(nueva_clase)
            await sesion_bd.commit()
                        
            time.sleep(20) 
            return f"Exito: Acabo de guardar la materia de {nombre_materia} el día {dia}."
    except Exception as error:
        return f"error al guardar {nombre_materia}: {str(error)}"

async def leer_horario_con_ia(ruta_de_la_foto: str):
    print("\nProcesando tu horario. Por favor espera...")
    
    _, extension = os.path.splitext(ruta_de_la_foto.lower())
    tipo_mime = "image/jpeg" if extension in ['.jpg', '.jpeg'] else f"image/{extension[1:]}"
    
    cerebro_ia = ChatGoogleGenerativeAI(
        model="gemini-2.5-flash", 
        temperature=0.1, 
        max_retries=1
    )  
    
    with open(ruta_de_la_foto, "rb") as archivo_foto:
        foto_en_texto = base64.b64encode(archivo_foto.read()).decode('utf-8')

    instrucciones = """
    Eres un asistente experto en lectura de horarios universitarios/escolares.
    Analiza la imagen adjunta y extrae TODAS las clases.
    REGLA PARA LOS DÍAS: Fíjate en las columnas (L, M, I, J, V, S). Extrae el nombre del día completo:
    L = Lunes, M = Martes, I = Miércoles, J = Jueves, V = Viernes, S = Sábado.
    Devuelve ÚNICAMENTE un objeto JSON válido con la siguiente estructura exacta:
    {
      "clases": [
        {
          "nombre_materia": "Nombre de la materia",
          "dia": "Día de la semana completo (Ej. Miércoles)",
          "inicio_fecha_hora": "YYYY-MM-DDTHH:MM:SSZ",
          "fin_fecha_hora": "YYYY-MM-DDTHH:MM:SSZ",
          "nombre_salon": "Nombre del salón o aula",
          "nombre_maestro": "Nombre del profesor"
        }
      ]
    }
    No incluyas markdown, solo el texto JSON puro. Si no hay salón o maestro, pon 'Sin maestro' o 'Sin salon'.
    """

    mensaje = [
        {"type": "text", "text": instrucciones},
        {"type": "image_url", "image_url": {"url": f"data:{tipo_mime};base64,{foto_en_texto}"}}
    ]

    exito = False
    intentos = 0
    max_intentos = 3

    while not exito and intentos < max_intentos:
        try:
            respuesta = await cerebro_ia.ainvoke([HumanMessage(content=mensaje)])
            texto_respuesta = respuesta.content.strip()
            
            if texto_respuesta.startswith("```json"):
                texto_respuesta = texto_respuesta[7:-3]
            elif texto_respuesta.startswith("```"):
                texto_respuesta = texto_respuesta[3:-3]
                
            datos_extraidos = json.loads(texto_respuesta)
            lista_clases = datos_extraidos.get("clases", [])
            
            print(f"\n¡Extracción exitosa! Se encontraron {len(lista_clases)} clases.")
            print("Guardando en la base de datos...\n")
            
            for clase in lista_clases:
                resultado = await guardar_clase_en_postgres(
                    nombre_materia=clase.get("nombre_materia"),
                    dia=clase.get("dia", "Sin día"), 
                    inicio_fecha_hora=clase.get("inicio_fecha_hora"),
                    fin_fecha_hora=clase.get("fin_fecha_hora"),
                    nombre_salon=clase.get("nombre_salon", "Sin salon"),
                    nombre_maestro=clase.get("nombre_maestro", "Sin maestro")
                )
                print(f"- {resultado}")
                
            exito = True 
            
        except Exception as e:
            error_str = str(e)
            if "429" in error_str or "RESOURCE_EXHAUSTED" in error_str:
                intentos += 1
                print(f"\n[Aviso de Cuota] Google está procesando la info. Pausando automáticamente por 60 segundos... (Intento {intentos}/{max_intentos})")
                await asyncio.sleep(60)
            else:
                print(f"\nError al procesar el horario: {e}")
                break

    print("\n¡El procesamiento de tu horario ha terminado!")

async def actualizar_clases_en_bd(datos_modificados):
    """ Toma los datos recopilados de la tabla y los sube a la BD de forma asíncrona """
    async with crear_sesion() as sesion_bd:
        for clase_dict in datos_modificados:
            consulta = select(ClaseDelHorario).where(ClaseDelHorario.id == clase_dict["id"])
            resultado = await sesion_bd.execute(consulta)
            clase_obj = resultado.scalar_one_or_none()
            
            if clase_obj:
                clase_obj.materia = clase_dict["materia"]
                clase_obj.dia = clase_dict["dia"]
                clase_obj.profesor = clase_dict["profesor"]
                clase_obj.contacto_profesor = clase_dict["contacto"]
                clase_obj.salon = clase_dict["salon"]
                
                # Transformamos el texto HH:MM de vuelta a la hora del sistema
                try:
                    h_ini, m_ini = map(int, clase_dict["inicio"].split(":"))
                    if clase_obj.hora_inicio:
                        clase_obj.hora_inicio = clase_obj.hora_inicio.replace(hour=h_ini, minute=m_ini)
                        
                    h_fin, m_fin = map(int, clase_dict["fin"].split(":"))
                    if clase_obj.hora_fin:
                        clase_obj.hora_fin = clase_obj.hora_fin.replace(hour=h_fin, minute=m_fin)
                except Exception:
                    pass
        await sesion_bd.commit()

def recopilar_datos_tabla(tabla):
    """ Extrae toda la información actual de las filas de la tabla para guardarla """
    datos_recolectados = []
    for fila_id in tabla.get_children():
        fila_datos = tabla.item(fila_id, 'values')
        datos_recolectados.append({
            "id": fila_datos[0],
            "materia": fila_datos[1],
            "dia": fila_datos[2],
            "profesor": fila_datos[3],
            "contacto": fila_datos[4], # Recuperamos el contacto oculto modificado
            "salon": fila_datos[5],
            "inicio": fila_datos[6],
            "fin": fila_datos[7]
        })
    return datos_recolectados

            btn_eliminar = ttk.Button(fila_frame, text="❌", width=3, command=eliminar_fila)
            btn_eliminar.pack(side=tk.LEFT, padx=5)
            
            filas_widgets.append((combo_tipo, ent_valor, fila_frame))

        # Cargamos contactos previos
        try:
            contactos_guardados = json.loads(valores_actuales[4])
            if not isinstance(contactos_guardados, list): contactos_guardados = []
        except:
            if valores_actuales[4] and valores_actuales[4] not in ("Sin contacto", "[]"):
                contactos_guardados = [{"tipo": "Otro", "valor": valores_actuales[4]}]
            else:
                contactos_guardados = []
                
        for c in contactos_guardados:
            agregar_fila_contacto(c.get("tipo", ""), c.get("valor", ""))
            
        # Si no había contactos, ponemos un campo vacío por defecto
        if not contactos_guardados:
            agregar_fila_contacto()
            
        # Botón para añadir un nuevo medio de contacto
        btn_add_contacto = ttk.Button(form_frame, text="➕ Agregar otro medio", command=agregar_fila_contacto)
        btn_add_contacto.grid(row=4, column=1, sticky=tk.W, pady=(0,10), padx=10)
        
        # --- FIN ZONA CONTACTOS ---
        
        # Resto de Entradas Clásicas
        campos_restantes = [
            ("Salón:", 5, 5), ("Hora de Inicio (HH:MM):", 6, 6), ("Hora de Fin (HH:MM):", 7, 7)
        ]
        for etiqueta, col_idx, grid_row in campos_restantes:
            ttk.Label(form_frame, text=etiqueta).grid(row=grid_row, column=0, sticky=tk.W, pady=5)
            ent = ttk.Entry(form_frame, width=35)
            ent.insert(0, valores_actuales[col_idx])
            ent.grid(row=grid_row, column=1, pady=5, padx=10, sticky=tk.W)
            entradas[col_idx] = ent
            
        def confirmar_edicion_local():
            if len(entradas[6].get().split(":")) != 2 or len(entradas[7].get().split(":")) != 2:
                messagebox.showerror("Error de Formato", "Las horas deben estar en formato HH:MM (Ejemplo: 14:30)")
                return
                
            # Procesamos todos los contactos llenos
            nuevos_contactos = []
            for combo, ent, _ in filas_widgets:
                t = combo.get()
                v = ent.get().strip()
                if v: # Solo guardamos si escribió algo
                    nuevos_contactos.append({"tipo": t, "valor": v})
                    
            json_contactos_final = json.dumps(nuevos_contactos)

            # Actualizamos la tabla
            tabla.item(item_id, values=(
                valores_actuales[0], # ID oculto
                entradas[1].get(),   # Materia
                entradas[2].get(),   # Dia
                entradas[3].get(),   # Maestro
                json_contactos_final,# Contacto (Se guarda oculto y formateado)
                entradas[5].get(),   # Salon
                entradas[6].get(),   # Inicio
                entradas[7].get()    # Fin
            ))
            
            messagebox.showinfo("Listo", "Cambios aplicados temporalmente. No olvides darle a 'Guardar y Actualizar' en la ventana principal.", parent=ventana_edicion)
            ventana_edicion.destroy()

        ttk.Button(ventana_edicion, text=" Aceptar Cambios", command=confirmar_edicion_local).pack(pady=20)


    # --- SISTEMA DE DOBLE CLIC INTELIGENTE (Requerimiento 2) ---
    def al_doble_clic(event):
        item_id = tabla.identify_row(event.y)
        columna_id = tabla.identify_column(event.x)
        
        if not item_id:
            return
            
        # Si la columna clickeada es la #4 (Maestro), mostramos los detalles
        if columna_id == '#4':
            valores = tabla.item(item_id, 'values')
            mostrar_contactos_maestro(valores[3], valores[4])
        else:
            # Si se hace doble clic en cualquier otra parte, abrimos la edición
            abrir_pestaña_editar()

    tabla.bind("<Double-1>", al_doble_clic)


    # --- ZONA DE BOTONES DE ACCIÓN (Requerimientos 3 y 4) ---
    def solicitar_guardado():
        # Enciende el interruptor. El bucle asíncrono lo atrapará sin cerrar la ventana.
        estado_ventana["guardar_ahora"] = True

    def solicitar_cierre():
        # Enciende el interruptor de cierre. Rompe el bucle y cierra la ventana.
        estado_ventana["cerrar"] = True
        
    # Atrapamos si se pica la 'X' de la esquina
    ventana.protocol("WM_DELETE_WINDOW", solicitar_cierre)

    marco_botones = ttk.Frame(ventana)
    marco_botones.pack(pady=15)
    
    btn_editar = ttk.Button(marco_botones, text=" Editar Selección", command=abrir_pestaña_editar)
    btn_editar.grid(row=0, column=0, padx=10)
    
    btn_guardar = ttk.Button(marco_botones, text=" Guardar y Actualizar", command=solicitar_guardado)
    btn_guardar.grid(row=0, column=1, padx=10)
    
    btn_cerrar = ttk.Button(marco_botones, text=" Cerrar y volver a Terminal", command=solicitar_cierre)
    btn_cerrar.grid(row=0, column=2, padx=10)

    # Devolvemos los objetos para controlarlos asincronamente
    return ventana, tabla, estado_ventana


async def extraer_y_mostrar_interfaz():
    global usuario_actual_id
    print("\nAbriendo tu ventana interactiva...")
    
    async with crear_sesion() as sesion_bd:
        consulta = select(ClaseDelHorario).where(ClaseDelHorario.usuario_id == usuario_actual_id)
        resultado = await sesion_bd.execute(consulta)
        todas_las_clases = resultado.scalars().all()

    if not todas_las_clases:
        print("No se encontraron clases para mostrar. Asegurate de que tienes datos guardados.")
        return

    # Creamos la ventana de forma no-bloqueante
    ventana, tabla, estado_ventana = crear_ventana_interactiva(todas_las_clases)
    
    try:
        # BUCLE PRINCIPAL (Mantiene la ventana viva y escucha botones sin congelar Python)
        while not estado_ventana["cerrar"]:
            ventana.update()
            
            # Requerimiento 3: Si se solicitó guardar, se hace y NO se rompe el bucle
            if estado_ventana["guardar_ahora"]:
                estado_ventana["guardar_ahora"] = False
                print("Subiendo cambios a la base de datos...")
                
                datos_nuevos = recopilar_datos_tabla(tabla)
                await actualizar_clases_en_bd(datos_nuevos)
                
                messagebox.showinfo("Éxito", "¡Tus cambios han sido guardados en la base de datos exitosamente!", parent=ventana)
                print("¡Actualización completada!")
                
            await asyncio.sleep(0.05) # Pausa mínima para no saturar el CPU
            
    except tk.TclError:
        pass # Ignoramos errores si la ventana se fuerza a cerrar con la X
    finally:
        # Requerimiento 4: Solo cuando se rompe el bucle (cerrar=True), se destruye la ventana
        try:
            ventana.destroy()
            print("Saliendo......")
        except:
            pass
